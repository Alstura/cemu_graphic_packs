int clampFI32(int v)
{
if( v == 0x7FFFFFFF )
	return floatBitsToInt(1.0);
else if( v == 0xFFFFFFFF )
	return floatBitsToInt(0.0);
return floatBitsToInt(clamp(intBitsToFloat(v), 0.0, 1.0));
}
float mul_nonIEEE(float a, float b){ if( a == 0.0 || b == 0.0 ) return 0.0; return a*b; }
void main()
{
vec4 R0f = vec4(0.0);
vec4 R1f = vec4(0.0);
vec4 R2f = vec4(0.0);
vec4 R3f = vec4(0.0);
vec4 R127f = vec4(0.0);
float backupReg0f, backupReg1f, backupReg2f, backupReg3f, backupReg4f;

R127f.x = mul_nonIEEE(R3f.z, R3f.z);
PV0f.y = -(R0f.w) + R3f.w;
R127f.z = mul_nonIEEE(R3f.y, R3f.y);

	float rred = $Red / 255.0f;

	float luma = dot(passPixelColor0.rgb, vec3(0.2126, 0.7152, 0.0722));

	luma = luma*$Bright;

	passPixelColor0 = vec4(luma*rred, luma*ggreen, luma*bblue, R0f.w*$Bright);
}
else
{
